初始化一些路径

FILE = Path(__file__).resolve()#得到当前这个detect.py的绝对路径
ROOT = FILE.parents[0]  # YOLOv5 root directory FILE的父目录，就是yolov5的路径
if str(ROOT) not in sys.path:#确保后面导入的包能够正常导入
    sys.path.append(str(ROOT))  # add ROOT to PATH
ROOT = Path(os.path.relpath(ROOT, Path.cwd()))  # relative，讲ROOT转换为相对路径
